381  # Copiando uma lista para outra (Shallow Copy e Deep Copy) 
382   
383  # Forma 1 - Deep Copy 
384   
385  lista = [1, 2, 3] 
386  print(f'lista -> {lista}') 
387   
388  nova = lista.copy() 
389  print(f'nova -> {nova}') 
390   
391  nova.append(4) 
392   
393  print(f'lista -> {lista}') 
394  print(f'nova -> {nova}') 
395   
396  # Veja que ao utilizarmos lista.copy() copiamos os dados da lista para uma nova lista, mas elas ficaram 
397  # totalmente independentes, ou seja, modificando uma lista, não afeta a outra. Isso em Python é chamado de 
398  # Deep Copy (cópia profunda) 
399  """
400  
401  # Forma 2 - Shallow Copy
402  
403  lista = [1, 2, 3]
404  print(f'lista -> {lista}')
405  
406  nova = lista  # cópia
407  print(f'nova -> {nova}')
408  
409  nova.append(4)
410  
411  print(f'lista -> {lista}')
412  print(f'nova -> {nova}')
413  
414  # Veja que utilizamos a cópia via atribuição e copiamos os dados da lista para a nova lista, mas
